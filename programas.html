Operações Básicas

Realize algumas operações matemáticas simples:

# Soma
a <- 5
b <- 3
soma <- a + b
print(soma)

# Subtração
subtracao <- a - b
print(subtracao)

# Multiplicação
multiplicacao <- a * b
print(multiplicacao)

# Divisão
divisao <- a / b
print(divisao)

Criação de Vetores

Crie e manipule vetores:

# Criar um vetor
numeros <- c(1, 2, 3, 4, 5)

# Acessar elementos do vetor
print(numeros[1])  # Primeiro elemento
print(numeros[2:4])  # Do segundo ao quarto elemento

# Soma dos elementos do vetor
soma_numeros <- sum(numeros)
print(soma_numeros)

Funções Simples

Defina e use uma função simples:

# Definir uma função
soma <- function(a, b) {
  return(a + b)
}

# Usar a função
resultado <- soma(4, 7)
print(resultado)

Gráfico Básico

Crie um gráfico básico com a função plot:

# Criar dados
x <- c(1, 2, 3, 4, 5)
y <- c(2, 3, 5, 7, 11)

# Criar gráfico
plot(x, y, type="b", col="blue", main="Gráfico de exemplo", xlab="Eixo X", ylab="Eixo Y")
